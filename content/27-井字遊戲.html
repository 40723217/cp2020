def diagonal_match(game):
	if game[1][1] != 0:
		if game[1][1] == game[0][0] == game[2][2]: 
			return game[1][1]
		elif game[1][1] == game[0][2] == game[2][0]:
			return game[1][1]			
	return 0
if line_match(game) > 0:
